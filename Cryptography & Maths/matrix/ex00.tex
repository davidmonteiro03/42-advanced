\section{Add Vectors}
\[
\vec{u} + \vec{v} =
\begin{bmatrix}
	u_1 \\
	u_2 \\
	\dots \\
	u_n
\end{bmatrix} +
\begin{bmatrix}
	v_1 \\
	v_2 \\
	\dots \\
	v_n
\end{bmatrix} =
\begin{bmatrix}
	u_1 + v_1 \\
	u_2 + v_2 \\
	\dots \\
	u_n + v_n
\end{bmatrix}
\]
\section{Subtract Vectors}
\[
\vec{u} - \vec{v} =
\begin{bmatrix}
	u_1 \\
	u_2 \\
	\dots \\
	u_n
\end{bmatrix} -
\begin{bmatrix}
	v_1 \\
	v_2 \\
	\dots \\
	v_n
\end{bmatrix} =
\begin{bmatrix}
	u_1 - v_1 \\
	u_2 - v_2 \\
	\dots \\
	u_n - v_n
\end{bmatrix}
\]
\section{Scale Vector}
\[
k\vec{u} =
k
\begin{bmatrix}
	u_1 \\
	u_2 \\
	\dots \\
	u_n
\end{bmatrix}=
\begin{bmatrix}
	ku_1 \\
	ku_2 \\
	\dots \\
	ku_n
\end{bmatrix}
\]
